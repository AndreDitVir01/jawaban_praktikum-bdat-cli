SET SERVEROUTPUT ON
DEFINE p_country_id='CA'
SET VERIFY OFF
DECLARE
TYPE record_type_country is record
	(country_id CHAR(2), country_name VARCHAR2(50));
record_country record_type_country;
v_country_id CHAR(2);
BEGIN
	FOR i in 1..4 LOOP
	IF i=1 THEN
		v_country_id = '&p_country_id';
	ELSEIF i=2 THEN
		v_country_id = 'DE';
	ELSEIF i=3 THEN 
		v_country_id = 'UK';
	ELSEIF i=4 THEN
		v_country_id = 'US';
	END IF;
	SELECT country_id, country_name
	INTO record_country.country_id record_country.country_name
	FROM countries
	WHERE country_id = v_country_id;
	DBMS_OUTPUT.PUT_LINE(record_country.country_id |''|
	record_country.country_name);
	END LOOP;
END;
/

SET SERVEROUTPUT ON
DECLARE
TYPE dept_table_type is table of department_name%TYPE
INDEX BY BINARY_INTEGER;
my_dept_table dept_table_type;
v_deptno departments.department_id%TYPE;
BEGIN
	FOR i in 1..7 LOOP
	IF i=1 THEN v_deptno = 10;
	ELSEIF i=2 THEN v_deptno = 20;
	ELSEIF i=3 THEN v_deptno = 50;
	ELSEIF i=4 THEN v_deptno = 60;
	ELSEIF i=5 THEN v_deptno = 80;
	ELSEIF i=6 THEN v_deptno = 90;
	ELSEIF i=7 THEN v_deptno = 110;
	END IF;
	SELECT department_name
	INTO my_dept_table(i);
	FROM departments
	WHERE department_id = v_deptno;
	END LOOP;
	FOR i in 1..7 LOOP
		DBMS_OUTPUT.PUT_LINE(my_dept_table(i));
	END LOOP;
END;
